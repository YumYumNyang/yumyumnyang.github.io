            numStr = "";

            numStr = numStr + divide.toString();

}

function checkPrimeNumber(num){
    if(num <= 1) return false;
    if(num === 2) return true;
    for(let i = 2; i <= parseInt(Math.sqrt(num)); i++){
        if(num%i === 0) return false;
    }
    return true;
}
